// --- Data Processing Module ---
namespace SecureAnalytics
{
    public class DataProcessor
    {
        public List<double> Data { get; private set; }

        public DataProcessor(IEnumerable<double> initialData)
        {
            Data = new List<double>(initialData);
            Console.WriteLine("Processor Initialized.");
        }

        /// <summary>
        /// Calculates the average value of the dataset.
        /// </summary>
        public double CalculateAverage()
        {
            if (Data.Count == 0)
                return 0;

            double sum = 0;
            foreach (var value in Data)
            {
                sum += value;
            }
            return sum / Data.Count;
        }
    }
}
